Task # 2 (own). Динамическое отображение пароля

Создайте поле для ввода пароля с возможностью его динамического отображения (показать/скрыть пароль). Кнопка должна менять иконку и состояние поля пароля между текстовым и скрытым вводом.

const buttonTogglePassword = document.body.querySelector( "[data-toggle-password]" ); function togglePasswordVisibility() { const password = document.body.querySelector("#password"); if (password.value.length && password.type === "password") { password.type = "text"; buttonTogglePassword.textContent = "Hide Password"; } else { password.type = "password"; buttonTogglePassword.textContent = "Show Password"; } }
